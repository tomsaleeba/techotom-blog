$('#popupBlockerCheckButton').click(function(){
$('#popupCheckMsg').html('');
setTimeout(function(){
// var popupWindow = window.open('popupCheck', '_blank');
var url = '${pageContext.request.contextPath}/popupCheck';
var popupWindow = window.open(url, '_blank', "location=no,toolbar=no,menubar=no,status=no,scrollbars=yes,resizable=no,width=900,height=330", true);
if (popupWindow == null || typeof(popupWindow)=='undefined') {
$('#popupCheckMsg').html('Please disable your pop-up blocker and click the "Open" link again.');
}
else {
$('#popupCheckMsg').html('yay');
}
}, 1100);
});

Pop ups must be fired by a click and <1 second after the click event. After that they are caught by the blocker.
